327 UInt32 SubWord(const UInt32 word) {

341 int KeyExpansion(const UChar key[], UInt32 **word, const int keySize) {
342  int numRounds, numCol, cnt;
343  UInt32 tmp;
344 
345 
346  switch (keySize) { // broj rundi u zavisnosti od kljuca
347  case 128: numRounds = 10; numCol = 4; break;
348  case 192: numRounds = 12; numCol = 6; break;
349  case 256: numRounds = 14; numCol = 8; break;
350  default: return 1; // nije dobra duzina kljuca
351  }
352 
353  if ((*word = malloc((NUM_ROW * numRounds + NUM_ROW) * sizeof(UInt32*))) == NULL) {
354  fprintf(stderr, "Nedovoljno memorije\n");
355  return -1;
356  }
357 
358  for (cnt = 0; cnt < numCol; cnt++)
359  (*word)[cnt] = UCharToUInt32(key + NUM_ROW * cnt); // stvaranje reci od niza karaktera
360 
361  for (cnt = numCol; cnt < NUM_ROW * (numRounds + 1); cnt++) {
362  tmp = (*word)[cnt - 1];
363  if ((cnt % numCol) == 0) {
364  RotWord(&tmp);
365  tmp = SubWord(tmp) ^ ROUND_CONSTANT[(cnt - 1) / numCol];
366  }
367  else if (numCol > 6 && cnt % numCol == 4)
368  tmp = SubWord(tmp);
369  (*word)[cnt] = (*word)[cnt - numCol] ^ tmp;
370  }
371  return 0; // sve je dobro proslo
372 }
